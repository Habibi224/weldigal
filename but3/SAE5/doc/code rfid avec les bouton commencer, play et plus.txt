// Convertir l'UID en format texte
String getUIDString() {
  String uid = "";
  for (byte i = 0; i < rfid.uid.size; i++) {
    if (rfid.uid.uidByte[i] < 0x10) uid += "0";
    uid += String(rfid.uid.uidByte[i], HEX);
    if (i < rfid.uid.size - 1) uid += ":";
  }
  uid.toUpperCase();
  return uid;
}

void handleRobots() {
 

  String json = "{\"A\":\"" + (robotA == "" ? "Badger" : robotA) +

}


void setup() {
    Serial.begin(9600);

  delay(1000);

  pinMode(LED_RFID, OUTPUT);
  digitalWrite(LED_RFID, LOW);



  // RFID
  SPI.begin(18, 19, 23, SS_PIN);
  rfid.PCD_Init();


 // WIFI
  WiFi.mode(WIFI_STA);
  WiFi.begin(ssid, password);
  while (WiFi.status() != WL_CONNECTED) delay(500);

  server.on("/", handleRoot);
  server.on("/robots", handleRobots);


  server.on("/start", []() {
  competitionName = server.arg("name");
  rfidActive = true;
  robotA = "";
  robotB = "";
  digitalWrite(LED_RFID, HIGH);
  server.send(200, "text/plain", "STARTED");
});

server.on("/stop", []() {
  rfidActive = false;
  digitalWrite(LED_RFID, LOW);
  server.send(200, "text/plain", "STOPPED");
});


server.on("/reset", []() {
  robotA = "";
  robotB = "";
  competitionName = "";

  rfidActive = false;
  digitalWrite(LED_RFID, LOW);

  server.send(200, "text/plain", "RESET");
});

  // Démarrage du serveur Web
  server.begin();
}

void loop() {
  server.handleClient();

// ---- RFID : nouvelle carte ? ----
// Si RFID désactivé → on ignore
if (!rfidActive) {
  return;
}

if (!rfid.PICC_IsNewCardPresent() || !rfid.PICC_ReadCardSerial())
   
return;


  String uid = getUIDString();
  Serial.println(uid);

  // ----- Associer UID à nom -----
  String nom = "";
  if (uid == "C3:83:86:E2") nom = "alpha";
  else if (uid == "21:6D:51:5D") nom = "beta";
  else if (uid == "E9:DF:08:83") nom = "gamma";
  else if (uid == "A3:BE:FA:2C") nom = "delta";
  else if (uid == "23:9B:3F:13") nom = "epsilon";
  else nom = "Inconnu";

  // -------------------------
  // LOGIQUE D'AFFECTATION
  // -------------------------

  if (robotA == "") {
    robotA = nom;
  }
  else if (robotB == "" && nom != robotA) {
    robotB = nom;
  }
  else {
  }

  rfid.PICC_HaltA();
  rfid.PCD_StopCrypto1();
}
