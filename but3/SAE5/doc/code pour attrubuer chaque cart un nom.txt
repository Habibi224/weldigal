#include <SPI.h>
#include <MFRC522.h>

// Pins RC522 -> ESP32
#define SS_PIN   5
#define RST_PIN  33

MFRC522 rfid(SS_PIN, RST_PIN);

String getUIDString() {
  String uid = "";
  for (byte i = 0; i < rfid.uid.size; i++) {
    if (rfid.uid.uidByte[i] < 0x10) uid += "0";
    uid += String(rfid.uid.uidByte[i], HEX);
    if (i < rfid.uid.size - 1) uid += ":";
  }
  uid.toUpperCase();
  return uid;
}

void setup() {
  Serial.begin(9600);
  SPI.begin(18, 19, 23, SS_PIN);
  rfid.PCD_Init();
  Serial.println("Lecteur RFID prêt. Passe une carte...");
}

void loop() {
  if (!rfid.PICC_IsNewCardPresent() || !rfid.PICC_ReadCardSerial()) {
    delay(10);
    return;
  }

  String uid = getUIDString();
  Serial.print("UID Carte: ");
  Serial.println(uid);

  // --------------------------
  // ASSOCIER LES UID AUX NOMS
  // --------------------------

  if (uid == "C3:83:86:E2") {
    Serial.println("alpha");
  }
  else if (uid == "21:6D:51:5D") {
    Serial.println("beta");
  }
  else if (uid == "E9:DF:08:83") {
    Serial.println("gamma");
  }
  else if (uid == "A3:BE:FA:2C") {
    Serial.println("delta");
  }
  else if (uid == "23:9B:3F:13") {
    Serial.println("epsilon");
  }
  else {
    Serial.println("Carte inconnue !");
  }

  rfid.PICC_HaltA();
  rfid.PCD_StopCrypto1();
}
