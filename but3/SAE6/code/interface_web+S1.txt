// --------- PWM S1 ----------
#define S1_PIN 23
#define S1_CHANNEL 0
#define S1_RESOLUTION 10      // 10 bits -> 0 à 1023
uint32_t S1_FREQ = 5000;      // Fréquence par défaut (Hz)
uint8_t S1_DUTY_PERCENT = 50; // Duty-cycle par défaut (%)

// --------- Serveur Web ----------
WebServer server(80);

// --------- Fonction pour mettre à jour S1 ----------
void updateS1PWM()
{
  // Duty-cycle calculé
  uint32_t duty = (S1_DUTY_PERCENT * ((1 << S1_RESOLUTION) - 1)) / 100;

  // Mise à jour fréquence et duty
  ledcSetup(S1_CHANNEL, S1_FREQ, S1_RESOLUTION);
  ledcWrite(S1_CHANNEL, duty);
}

// --------- Gestion requête racine ---------
void handleRoot()
{
  server.send(200, "text/html", htmlPage);
}

// --------- Gestion envoi des valeurs S1 ---------
void handleSetS1() {
  if (server.hasArg("freq")) {
    uint32_t f = server.arg("freq").toInt();
    if (f >= 1000 && f <= 1000000) {
      S1_FREQ = f;
    }
  }

  if (server.hasArg("duty")) {
    uint8_t d = server.arg("duty").toInt();
    if (d >= 1 && d <= 99) {
      S1_DUTY_PERCENT = d;
    }
  }

  updateS1PWM();

  server.send(200, "text/plain", "OK");
}


void setup()
{
  Serial.begin(115200);

  // --------- PWM S1 initial ----------
  ledcSetup(S1_CHANNEL, S1_FREQ, S1_RESOLUTION);
  ledcAttachPin(S1_PIN, S1_CHANNEL);
  updateS1PWM();

  // --------- Connexion WiFi ----------
  WiFi.begin(ssid, password);
  while (WiFi.status() != WL_CONNECTED)
  {
    delay(500);
    Serial.print(".");
  }

  // --------- Routes serveur ---------
  server.on("/", handleRoot);
  server.on("/setS1", handleSetS1);
  server.begin();
}

void loop()
{
  server.handleClient();
}
