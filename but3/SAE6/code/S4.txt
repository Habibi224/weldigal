// --------- Fonctions PWM ---------
void applyPWMConfig()
{

  q = 0.5 * n + 1;
  pwmFreq_S3 = (uint32_t)(q * pwmFreq);

  // recalculer ticks
  const uint32_t ticksPerPeriod = (1u << PWM_RESOLUTION); // ex: 1024
  uint32_t dutyTicks = (uint32_t)dutyPercent * (ticksPerPeriod - 1) / 100u;
  uint32_t hpoint = (uint32_t)phaseDeg * ticksPerPeriod / 360u; // hpoint en ticks

  uint32_t hpoint_S4 = (uint32_t)phaseDeg_S4 * ticksPerPeriod / 360u;
  hpoint_S4 %= ticksPerPeriod;

  // s'assure que hpoint est dans [0, ticksPerPeriod-1]
  hpoint %= ticksPerPeriod;

  // config des canaux (même fréquence, même résolution)
  ledcSetup(S1_CHANNEL, pwmFreq, PWM_RESOLUTION);
  ledcSetup(S2_CHANNEL, pwmFreq, PWM_RESOLUTION);
  ledcSetup(S3_CHANNEL, pwmFreq_S3, PWM_RESOLUTION);
  ledcSetup(S4_CHANNEL, pwmFreq_S3, PWM_RESOLUTION);

  // attacher pins (si pas encore attachées, ledcAttachPin est idempotent)
  ledcAttachPin(S1_PIN, S1_CHANNEL);
  ledcAttachPin(S2_PIN, S2_CHANNEL);
  ledcAttachPin(S3_PIN, S3_CHANNEL);
  ledcAttachPin(S4_PIN, S4_CHANNEL);

  // appliquer duty + hpoint (utilise l'API esp-idf)
  ledc_set_duty_with_hpoint(LEDC_HIGH_SPEED_MODE, S1_CHANNEL, dutyTicks, 0u);
  ledc_update_duty(LEDC_HIGH_SPEED_MODE, S1_CHANNEL);

  ledc_set_duty_with_hpoint(LEDC_HIGH_SPEED_MODE, S2_CHANNEL, dutyTicks, hpoint);
  ledc_update_duty(LEDC_HIGH_SPEED_MODE, S2_CHANNEL);

  ledc_set_duty_with_hpoint(LEDC_HIGH_SPEED_MODE, S3_CHANNEL, dutyTicks, hpoint);
  ledc_update_duty(LEDC_HIGH_SPEED_MODE, S3_CHANNEL);

  ledc_set_duty_with_hpoint(LEDC_HIGH_SPEED_MODE, S4_CHANNEL, dutyTicks, hpoint_S4);
  ledc_update_duty(LEDC_HIGH_SPEED_MODE, S4_CHANNEL);
}

// --------- Gestion requête racine ---------
void handleRoot()
{
  server.send_P(200, "text/html", htmlPage);
}

// --------- Gestion envoi des valeurs S1 (freq,duty,phase) ---------
void handleSetS1()
{
  bool changed = false;

  if (server.hasArg("freq"))
  {
    uint32_t f = (uint32_t)server.arg("freq").toInt();
    if (f >= 1000 && f <= 10000)
    { // bornes cohérentes avec ton UI
      pwmFreq = f;
      changed = true;
    }
  }

  if (server.hasArg("duty"))
  {
    uint8_t d = (uint8_t)server.arg("duty").toInt();
    if (d >= 1 && d <= 99)
    {
      dutyPercent = d;
      changed = true;
    }
  }

  if (server.hasArg("phase"))
  {
    uint16_t p = (uint16_t)server.arg("phase").toInt();
    if (p <= 359)
    {
      phaseDeg = p;
      changed = true;
    }
  }

  if (server.hasArg("phaseS4"))
  {
    uint16_t p4 = server.arg("phaseS4").toInt();
    if (p4 <= 360)
    {
      phaseDeg_S4 = p4;
      changed = true;
    }
  }

  if (server.hasArg("n"))
  {
    uint16_t val = server.arg("n").toInt();
    if (val <= 10)
    {
      n = val;
      changed = true;
    }
  }

  if (changed)
  {
    applyPWMConfig();
  }

  // Réponse utile au client (JSON-like simple)
  String resp = String("{freq:") + pwmFreq + ",duty:" + dutyPercent + ",phase:" + phaseDeg + "}";
  server.send(200, "text/plain", resp);
}

void setup()
{
  delay(100);

  // Démarre PWM avec valeur initiale (attache pins puis configure)
  ledcAttachPin(S1_PIN, S1_CHANNEL);
  ledcAttachPin(S2_PIN, S2_CHANNEL);
  ledcAttachPin(S3_PIN, S3_CHANNEL);
  ledcAttachPin(S4_PIN, S4_CHANNEL);
  applyPWMConfig();

  // ---------- WiFi ----------
  WiFi.begin(ssid, password);
  unsigned long start = millis();
  while (WiFi.status() != WL_CONNECTED)
  {
    delay(250);
    Serial.print(".");
    if (millis() - start > 15000)
    { // timeout 15s
      Serial.println("\nImpossible de se connecter au WiFi (timeout).");
      break;
    }
  }
  if (WiFi.status() == WL_CONNECTED)
  {
    Serial.println();
    Serial.print("WiFi OK, IP: ");
    Serial.println(WiFi.localIP());
  }

  // -------- Routes serveur ----------
  server.on("/", handleRoot);
  server.on("/setS1", handleSetS1);
  server.begin();
}

void loop()
{
  server.handleClient();
}
