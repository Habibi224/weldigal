#include <Arduino.h>
#include "driver/pcnt.h"

#define SIGNAL_GPIO 23
#define PCNT_UNIT   PCNT_UNIT_0
#define PCNT_CH     PCNT_CHANNEL_0

void setup() {
  // ----- Génération signal -----
  pinMode(LED_BUILTIN, OUTPUT);

  // ----- Serial -----
  Serial.begin(115200);

  // ----- Configuration PCNT -----
  pcnt_config_t pcnt_config = {};
  pcnt_config.pulse_gpio_num = SIGNAL_GPIO;   // GPIO à mesurer
  pcnt_config.ctrl_gpio_num = PCNT_PIN_NOT_USED;
  pcnt_config.unit = PCNT_UNIT;
  pcnt_config.channel = PCNT_CH;
  pcnt_config.pos_mode = PCNT_COUNT_INC;      // front montant
  pcnt_config.neg_mode = PCNT_COUNT_DIS;      // ignore front descendant
  pcnt_config.lctrl_mode = PCNT_MODE_KEEP;
  pcnt_config.hctrl_mode = PCNT_MODE_KEEP;
  pcnt_config.counter_h_lim = 32767;
  pcnt_config.counter_l_lim = 0;

  pcnt_unit_config(&pcnt_config);

  // Filtre anti-bruit
  pcnt_set_filter_value(PCNT_UNIT, 100);
  pcnt_filter_enable(PCNT_UNIT);

  pcnt_counter_pause(PCNT_UNIT);
  pcnt_counter_clear(PCNT_UNIT);
  pcnt_counter_resume(PCNT_UNIT);
}

void loop() {
  // ----- Génération signal carré ~1 kHz -----
  digitalWrite(LED_BUILTIN, HIGH);
  delayMicroseconds(500);
  digitalWrite(LED_BUILTIN, LOW);
  delayMicroseconds(500);

  // ----- Mesure fréquence PCNT -----
  static unsigned long t1 = 0;

  if (millis() - t1 >= 1000) {   
    t1 = millis();

    int16_t count = 0;
    pcnt_get_counter_value(PCNT_UNIT, &count);

    Serial.print("Frequence mesuree sur GPIO 23 : ");
    Serial.print(count);
    Serial.println(" Hz");
    Serial.println(t1);
    pcnt_counter_clear(PCNT_UNIT);
  }
}
